## Usage

```tsx
import {InputNumber} from "@/components/ui/input-number";
```

<br />
```tsx
const Preview = () => {
  const [number, setNumber] = useState("");
  return (
    <div className="flex flex-row space-x-3 items-center">
      <InputNumber
        placeholder="Number"
        className="mt-2"
        value={number}
        onChange={(value) => setNumber(value)}
      />
      <Button
        variant={"outline"}
        className="mt-2"
        onClick={() => alert(number)}
      >
        Submit
      </Button>
    </div>
  );
};
```

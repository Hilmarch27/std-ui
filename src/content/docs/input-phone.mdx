## Usage

```tsx
import {InputPhone} from "@/components/ui/input-phone";
```

<br />
```tsx
const Preview = () => {
  const [phone, setPhone] = useState("");
  return (
    <div className="flex flex-row space-x-3 items-center">
      <InputPhone
        placeholder="Phone"
        className="mt-2"
        value={phone}
        onChange={(value) => setPhone(value)}
      />

      <Button
        variant={"outline"}
        className="mt-2"
        onClick={() => alert(phone)}
      >
        Submit
      </Button>
    </div>
  );
};
```

### Adjust Country Code

You can adjust the country code with props

<br />
```tsx
<InputPhone
  placeholder="Phone"
  countryCode="65"
  value={phone}
  onChange={(value) => setPhone(value)}
/>
```
